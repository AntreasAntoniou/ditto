# Python dependencies for restoring + converting the on-device ogma models to
# CoreML — used by tools/restore-models.sh (which calls tools/_dl.py and
# tools/convert_ogma.py) and by the CI workflows. Target: Python 3.11.
#
# WHY torch is pinned: coremltools' Torch frontend only converts graphs traced
# by a torch version it has been tested against. The coremltools 8.x line here
# declares its ceiling as torch 2.5.0 ("Torch 2.5.0 is the most recent version
# that has been tested"); newer torch (2.7/2.12) traces ops coremltools can't
# lower and the convert dies (e.g. _cast "only 0-dimensional arrays..." on 2.12,
# a 'slice'/int dtype ValueError on 2.7). Pin torch to that tested ceiling.
#
# REPRODUCIBILITY: after your first green `Models` workflow run, freeze the rest
# of these to the resolved versions (copy from that run's `pip freeze`) so
# releases convert against a fully known-good stack and never drift.
torch==2.5.0
coremltools>=8.0,<9.0
transformers
sentencepiece
safetensors
huggingface_hub
