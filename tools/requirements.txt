# Python dependencies for restoring + converting the on-device ogma models to
# CoreML — used by tools/restore-models.sh (which calls tools/_dl.py and
# tools/convert_ogma.py) and by the CI workflows. Target: Python 3.11.
#
# These two pins are the EXACT versions that produced the shipping models
# (read from the .mlpackage metadata: com.github.apple.coremltools.version=9.0,
# source=torch==2.7.1, parity_cosine=1.00000). Do not loosen them blindly:
#  - Unpinned torch resolves to 2.12.x, which coremltools can't lower
#    (_cast: "only 0-dimensional arrays can be converted to Python scalars").
#  - coremltools 8.x hits a 'slice'-op regression on the model's rotary
#    embedding ("Could not convert Var to a NumPy dtype"). 9.0 converts cleanly.

coremltools==9.0
